# Filtering out rows with having missing values in the "Age" column.
missing_ages = titanic_df[titanic_df['Age'].isnull()]
# determine average age based on Sex and Pclass columns
mean_ages = titanic_df.groupby(['Sex','Pclass'])['Age'].mean()

def remove_missing_ages(row):
    # Replace missing age with the average age from the dataframe above (mean_ages)
    if pd.isnull(row['Age']):
        return mean_ages[row['Sex'],row['Pclass']]
    else:
        return row['Age']

titanic_df['Age'] =titanic_df.apply(remove_missing_ages, axis=1)

def bar_chart(feature):
    """ This functions takes the feature that we want to plot against survivors"""
    survived = titanic_df[titanic_df['Survived']==1][feature].value_counts()
    not_survived = titanic_df[titanic_df['Survived']==0][feature].value_counts()
    df = pd.DataFrame([survived,not_survived])
    df.index=['Survived','Not Survived']
    df.plot(kind='bar',stacked=False,title="Stacked Chart for  "+feature, figsize=(12,10))
